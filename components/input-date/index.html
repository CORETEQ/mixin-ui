Input Date

Used to enter or select a date value.

Import

import {
  XInputDate,
  XControl,
  XPopover,
  XCalendar
} from '@mixin-ui/kit';

Usage

@Component({
  template: `
    <x-input-date>
      <input x-control [(ngModel)]="value" />
      <x-calendar *x-popover />
    </x-input-date>
  `
})
export class Usage {
  readonly value = signal(new Date());
}

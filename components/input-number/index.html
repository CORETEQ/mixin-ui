Input Number

Used to enter a number, and increment or decrement the value using stepper buttons.

Import

import {
  XInputNumber,
  XControl
} from '@mixin-ui/kit';

Usage

@Component({
  template: `
    <x-input-number>
      <input x-control [(ngModel)]="value">
    </x-input-number>
  `
})
export class Usage {
  readonly value = signal(0);
}
